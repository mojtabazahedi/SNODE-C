132  void Transmitter::sendFrame(bool fin, uint8_t opCode, const char* payload, uint64_t payloadLength) {
133  uint64_t length = 0;
134 
135  if (payloadLength < 126) {
136  length = payloadLength;
137  } else if (payloadLength < 0x10000) {
138  length = 126;
139  } else {
140  length = 127;
141  }
142 
143  char header[2];
144  header[0] = static_cast<char>((fin ? 0b10000000 : 0) | opCode);
145

147 
148  sendFrameData(header, 2);

50  void DescriptorEventDispatcher::enable(DescriptorEventReceiver* eventReceiver, int fd) {
51  if (disabledEventReceiver[fd].contains(eventReceiver)) {
52  // same tick as disable
53  disabledEventReceiver[fd].remove(eventReceiver);
54  } else if (!eventReceiver->isEnabled() &&
55  (!enabledEventReceiver.contains(fd) || !enabledEventReceiver[fd].contains(eventReceiver))) {
56  // next tick as disable
57  enabledEventReceiver[fd].push_back(eventReceiver);
58  eventReceiver->enabled();
59  if (unobservedEventReceiver.contains(eventReceiver)) {
60  unobservedEventReceiver.remove(eventReceiver);
61  }
62  } else {
63  LOG(WARNING) << "EventReceiver double enable";
64  }
65  }

122  struct timeval DescriptorEventDispatcher::observeEnabledEvents() {
123  struct timeval nextTimeout = {LONG_MAX, 0};
124 
125  for (const auto& [fd, eventReceivers] : enabledEventReceiver) {
126  for (DescriptorEventReceiver* eventReceiver : eventReceivers) {
127  observedEventReceiver[fd].push_front(eventReceiver);
128  if (!eventReceiver->isSuspended()) {
129  fdSet.set(fd);
130  nextTimeout = std::min(nextTimeout, eventReceiver->getTimeout());
131  } else {
132  fdSet.clr(fd, true);
133  }
134  }
135  }
136  enabledEventReceiver.clear();
137 
138  return nextTimeout;
139  }

141  struct timeval DescriptorEventDispatcher::dispatchActiveEvents(struct timeval currentTime) {
142  struct timeval nextInactivityTimeout {
143  LONG_MAX, 0
144  };
145 
146  for (const auto& [fd, eventReceivers] : observedEventReceiver) {
147  DescriptorEventReceiver* eventReceiver = eventReceivers.front();
148  struct timeval maxInactivity = eventReceiver->getTimeout();
149  if (fdSet.isSet(fd)) {
150  eventCounter++;
151  eventReceiver->dispatchEvent();
152  eventReceiver->triggered(currentTime);
153  nextInactivityTimeout = std::min(nextInactivityTimeout, maxInactivity);
154  } else {
155  struct timeval inactivity = currentTime - eventReceiver->getLastTriggered();
156  if (inactivity >= maxInactivity) {
157  eventReceiver->timeoutEvent();
158  eventReceiver->disable();
159  } else {
160  nextInactivityTimeout = std::min(maxInactivity - inactivity, nextInactivityTimeout);
161  }
162  }
163  }
164 
165  return nextInactivityTimeout;
166  }

168  void DescriptorEventDispatcher::unobserveDisabledEvents() {
169  for (const auto& [fd, eventReceivers] : disabledEventReceiver) {
170  for (DescriptorEventReceiver* eventReceiver : eventReceivers) {
171  observedEventReceiver[fd].remove(eventReceiver);
172  if (observedEventReceiver[fd].empty() || observedEventReceiver[fd].front()->isSuspended()) {
173  if (observedEventReceiver[fd].empty()) {
174  observedEventReceiver.erase(fd);
175  }
176  fdSet.clr(fd, true);
177  } else {
178  fdSet.set(fd);
179  observedEventReceiver[fd].front()->triggered();
180  }
181  eventReceiver->disabled();
182  if (eventReceiver->observationCounter == 0) {
183  unobservedEventReceiver.push_back(eventReceiver);
184  }
185  eventReceiver->fd = -1;
186  }
187  }
188  disabledEventReceiver.clear();
189  }

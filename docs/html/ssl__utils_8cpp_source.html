57  SSL_CTX* ssl_ctx_new(const std::map<std::string, std::any>& options, bool server) {
58  std::string certChain;
59  std::string keyPEM;
60  std::string password;
61  std::string caFile;
62  std::string caDir;
63  bool useDefaultCADir = false;
64 
65  for (const auto& [name, value] : options) {
66  if (name == "certChain") {
67  certChain = std::any_cast<const char*>(value);
68  } else if (name == "keyPEM") {
69  keyPEM = std::any_cast<const char*>(value);
70  } else if (name == "password") {
71  password = std::any_cast<const char*>(value);
72  } else if (name == "caFile") {
73  caFile = std::any_cast<const char*>(value);
74  } else if (name == "caDir") {
75  caDir = std::any_cast<const char*>(value);
76  } else if (name == "useDefaultCADir") {
77  useDefaultCADir = std::any_cast<bool>(value);
78  }
79  }
80 
81  SSL_CTX* ctx = SSL_CTX_new(server ? TLS_server_method() : TLS_client_method());
82 
83  if (ctx != nullptr) {
84  bool sslErr = false;
85 
86  if (server) {
87  SSL_CTX_set_session_id_context(ctx, (const unsigned char*) &sslSessionCtxId, sizeof(sslSessionCtxId));
88  sslSessionCtxId++;
89  }
90  if (!caFile.empty() || !caDir.empty()) {
91  if (!SSL_CTX_load_verify_locations(
92  ctx, !caFile.empty() ? caFile.c_str() : nullptr, !caDir.empty() ? caDir.c_str() : nullptr)) {
93  ssl_log_error("Can not load CA file or non default CA directory");
94  sslErr = true;
95  }
96  }
97  if (!sslErr && useDefaultCADir) {
98  if (!SSL_CTX_set_default_verify_paths(ctx)) {
99  ssl_log_error("Can not load default CA directory");
100  sslErr = true;
101  }
102  }
103  if (!sslErr) {
104  if (server && (useDefaultCADir || !caFile.empty() || !caDir.empty())) {
105  SSL_CTX_set_verify_depth(ctx, 5);
106  SSL_CTX_set_verify(ctx, SSL_VERIFY_FLAGS, verify_callback);
107  }
108  if (!certChain.empty()) {
109  if (SSL_CTX_use_certificate_chain_file(ctx, certChain.c_str()) != 1) {
110  ssl_log_error("Can not load certificate chain");
111  sslErr = true;
112  } else if (!keyPEM.empty()) {
113  if (!password.empty()) {
114  SSL_CTX_set_default_passwd_cb(ctx, password_callback);
115  SSL_CTX_set_default_passwd_cb_userdata(ctx, ::strdup(password.c_str()));
116  }
117  if (SSL_CTX_use_PrivateKey_file(ctx, keyPEM.c_str(), SSL_FILETYPE_PEM) != 1) {
118  errno = EINVAL;
119  ssl_log_error("Can not load private key");
120  sslErr = true;
121  } else if (!SSL_CTX_check_private_key(ctx)) {
122  errno = EINVAL;
123  ssl_log_error("Private key not consistent with CA files");
124  sslErr = true;
125  }
126  }
127  }
128  }
129  if (sslErr) {
130  SSL_CTX_free(ctx);
131  ctx = nullptr;
132  }
133  } else {
134  ssl_log_error("SSL CTX not created");
135  }
136 
137  return ctx;
138  }

152  void ssl_log(const std::string& message, int sslErr) {
153  switch (sslErr) {
154  case SSL_ERROR_NONE:
155  [[fallthrough]];
156  case SSL_ERROR_ZERO_RETURN:
157  ssl_log_info(message);
158  break;
159  case SSL_ERROR_SYSCALL:
160  if (errno != 0) {
161  ssl_log_error(message);
162  }
163  break;
164  default:
165  ssl_log_warning(message);
166  break;
167  };
168  }

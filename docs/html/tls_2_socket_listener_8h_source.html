41  class SocketListener : public stream::SocketListener<stream::tls::SocketConnection<SocketT>> {
42  public:

47  SocketListener(const std::shared_ptr<const SocketContextFactory>& socketContextFactory,
48  const std::function<void(const SocketAddress&, const SocketAddress&)>& onConnect,
49  const std::function<void(SocketConnection*)>& onConnected,
50  const std::function<void(SocketConnection*)>& onDisconnect,
51  const std::map<std::string, std::any>& options)
52  : stream::SocketListener<SocketConnection>(
53  socketContextFactory,
54  onConnect,
55  [onConnected, this](SocketConnection* socketConnection) -> void {
56  SSL* ssl = socketConnection->startSSL(this->ctx);
57 
58  if (ssl != nullptr) {
59  SSL_CTX_set_tlsext_servername_arg(this->ctx, socketConnection);
60 
61  SSL_set_accept_state(ssl);
62 
63  socketConnection->doSSLHandshake(
64  [&onConnected, socketConnection](void) -> void { // onSuccess
65  LOG(INFO) << "SSL/TLS initial handshake success";
66  onConnected(socketConnection);
67  },
68  [](void) -> void { // onTimeout
69  LOG(WARNING) << "SSL/TLS initial handshake timed out";
70  },
71  [](int sslErr) -> void { // onError
72  ssl_log("SSL/TLS initial handshake failed", sslErr);
73  });
74  } else {
75  socketConnection->SocketConnection::SocketReader::disable();
76  ssl_log_error("SSL/TLS initialization failed");
77  }
78  },
79  [onDisconnect](SocketConnection* socketConnection) -> void { // onDisconnect
80  socketConnection->stopSSL();
81  onDisconnect(socketConnection);
82  },
83  options) {
84  ctx = ssl_ctx_new(options, true);
85  if (ctx != nullptr) {
86  SSL_CTX_set_tlsext_servername_callback(ctx, serverNameCallback);
87  }
88  }
89 
90  ~SocketListener() override {
91  ssl_ctx_free(ctx);
92  }
93 
94  void listen(const SocketAddress& localAddress, int backlog, const std::function<void(int)>& onError) {
95  if (ctx == nullptr) {
96  errno = EINVAL;
97  onError(errno);
98  stream::SocketListener<SocketConnection>::destruct();
99  } else {
100  stream::SocketListener<SocketConnection>::listen(localAddress, backlog, onError);
101  }
102  }
103 
104  protected:
105  static int serverNameCallback(SSL* ssl, [[maybe_unused]] int* al, void* arg) {
106  SocketConnection* socketConnection = static_cast<SocketConnection*>(arg);
107 
108  if (SSL_get_servername_type(ssl) != -1) {
109  socketConnection->serverNameIndication = SSL_get_servername(ssl, TLSEXT_NAMETYPE_host_name);
110  LOG(INFO) << "ServerNameIndication: " << socketConnection->serverNameIndication;
111  }
112 
113  return SSL_TLSEXT_ERR_OK;
114  }
115 
116  SSL_CTX* ctx = nullptr;

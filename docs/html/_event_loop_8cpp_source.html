58  void EventLoop::tick() {
59  struct timeval nextTimeout = readEventDispatcher.observeEnabledEvents();
60  nextInactivityTimeout = std::min(nextTimeout, nextInactivityTimeout);
61 
62  nextTimeout = writeEventDispatcher.observeEnabledEvents();
63  nextInactivityTimeout = std::min(nextTimeout, nextInactivityTimeout);
64 
65  nextTimeout = exceptionalConditionEventDispatcher.observeEnabledEvents();
66  nextInactivityTimeout = std::min(nextTimeout, nextInactivityTimeout);
67 
68  nextTimeout = timerEventDispatcher.getNextTimeout();
69  nextInactivityTimeout = std::min(nextTimeout, nextInactivityTimeout);
70 
71  int maxFd = readEventDispatcher.getMaxFd();
72  maxFd = std::max(writeEventDispatcher.getMaxFd(), maxFd);
73  maxFd = std::max(exceptionalConditionEventDispatcher.getMaxFd(), maxFd);
74 
75  if (maxFd >= 0 || !timerEventDispatcher.empty()) {
76  if (nextInactivityTimeout < timeval({0, 0})) {
77  nextInactivityTimeout = {0, 0};
78  }
79 
80  int counter = net::system::select(maxFd + 1,
81  &readEventDispatcher.getFdSet().get(),
82  &writeEventDispatcher.getFdSet().get(),
83  &exceptionalConditionEventDispatcher.getFdSet().get(),
84  &nextInactivityTimeout);
85 
86  if (counter >= 0) {
87  tickCounter++;
88  timerEventDispatcher.dispatch();
89 
90  struct timeval currentTime = {net::system::time(nullptr), 0};
91  nextInactivityTimeout = {LONG_MAX, 0};
92 
93  nextTimeout = readEventDispatcher.dispatchActiveEvents(currentTime);
94  nextInactivityTimeout = std::min(nextTimeout, nextInactivityTimeout);
95 
96  nextTimeout = writeEventDispatcher.dispatchActiveEvents(currentTime);
97  nextInactivityTimeout = std::min(nextTimeout, nextInactivityTimeout);
98 
99  nextTimeout = exceptionalConditionEventDispatcher.dispatchActiveEvents(currentTime);
100  nextInactivityTimeout = std::min(nextTimeout, nextInactivityTimeout);
101  } else if (errno != EINTR) {
102  PLOG(ERROR) << "select";
103  stop();
104  }
105  } else {
106  stop();
107  }
108 
109  readEventDispatcher.unobserveDisabledEvents();
110  writeEventDispatcher.unobserveDisabledEvents();
111  exceptionalConditionEventDispatcher.unobserveDisabledEvents();
112 
113  readEventDispatcher.releaseUnobservedEvents();
114  writeEventDispatcher.releaseUnobservedEvents();
115  exceptionalConditionEventDispatcher.releaseUnobservedEvents();
116  }

136  template <typename Request, typename Response>
137  void SocketContextT<Request, Response>::requestParsed() {
138  if (!requestInProgress) {
139  RequestContext& requestContext = requestContexts.front();
140 
141  requestInProgress = true;
142  if (requestContext.status == 0) {
143  if (((requestContext.request.connectionState == ConnectionState::Close) ||
144  (requestContext.request.httpMajor == 0 && requestContext.request.httpMinor == 9) ||
145  (requestContext.request.httpMajor == 1 && requestContext.request.httpMinor == 0 &&
146  requestContext.request.connectionState != ConnectionState::Keep) ||
147  (requestContext.request.httpMajor == 1 && requestContext.request.httpMinor == 1 &&
148  requestContext.request.connectionState == ConnectionState::Close))) {
149  requestContext.response.set("Connection", "close");
150  } else {
151  requestContext.response.set("Connection", "keep-alive");
152  }
153 
154  onRequestReady(requestContext.request, requestContext.response);
155  } else {
156  requestContext.response.status(requestContext.status).send(requestContext.reason);
157  terminateConnection();
158  }
159  }
160  }

162  template <typename Request, typename Response>
163  void SocketContextT<Request, Response>::sendToPeerCompleted() {
164  RequestContext& requestContext = requestContexts.front();
165 
166  // if 0.9 => terminate
167  // if 1.0 && (request != Keep || contentLength = -1) => terminate
168  // if 1.1 && (request == Close || contentLength = -1) => terminate
169  // if (request == Close) => terminate
170 
171  if ((requestContext.request.httpMajor == 0 && requestContext.request.httpMinor == 9) ||
172  (requestContext.request.httpMajor == 1 && requestContext.request.httpMinor == 0 &&
173  requestContext.request.connectionState != ConnectionState::Keep) ||
174  (requestContext.request.httpMajor == 1 && requestContext.request.httpMinor == 1 &&
175  requestContext.request.connectionState == ConnectionState::Close) ||
176  (requestContext.response.connectionState == ConnectionState::Close)) {
177  terminateConnection();
178  } else {
179  reset();
180 
181  requestContexts.pop_front();
182 
183  if (!requestContexts.empty() && requestContexts.front().ready) {
184  RequestContext& requestContext = requestContexts.front();
185 
186  requestInProgress = true;
187  if (requestContext.status == 0) {
188  onRequestReady(requestContext.request, requestContext.response);
189  } else {
190  requestContext.response.status(requestContext.status).send(requestContext.reason);
191  terminateConnection();
192  }
193  }
194  }

42  void Response::sendFile(const std::string& file, const std::function<void(int err)>& onError) {
43  std::string absolutFileName = file;
44 
45  if (std::filesystem::exists(absolutFileName)) {
46  std::error_code ec;
47  absolutFileName = std::filesystem::canonical(absolutFileName);
48 
49  if (std::filesystem::is_regular_file(absolutFileName, ec) && !ec) {
50  headers.insert({{"Content-Type", web::http::MimeTypes::contentType(absolutFileName)},
51  {"Last-Modified", httputils::file_mod_http_date(absolutFileName)}});
52  headers.insert_or_assign("Content-Length", std::to_string(std::filesystem::file_size(absolutFileName)));
53 
54  FileReader::connect(absolutFileName, *this, [this, onError](int err) -> void {
55  serverContext->terminateConnection();
56  onError(err);

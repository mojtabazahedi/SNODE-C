35  unsigned char* decode64(const char* input, int length) {
36  const int pl = 3 * length / 4;
37  unsigned char* output = reinterpret_cast<unsigned char*>(calloc(static_cast<std::size_t>(pl + 1), 1));
38  const int ol = EVP_DecodeBlock(output, reinterpret_cast<const unsigned char*>(input), length);
39  if (pl != ol) {
40  LOG(ERROR) << "Whoops, decode predicted " << pl << " but we got " << ol;
41  }
42  return output;
43  }

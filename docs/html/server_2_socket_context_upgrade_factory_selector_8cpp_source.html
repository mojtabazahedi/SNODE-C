110  SocketContextUpgradeFactory* SocketContextUpgradeFactorySelector::load(const std::string& filePath) {
111  SocketContextUpgradeFactory* socketContextUpgradeFactory = nullptr;
112 
113  void* handle = dlopen(filePath.c_str(), RTLD_LAZY | RTLD_LOCAL);
114 
115  if (handle != nullptr) {
116  SocketContextUpgradeFactory* (*plugin)() = reinterpret_cast<SocketContextUpgradeFactory* (*) ()>(dlsym(handle, "plugin"));
117 
118  if (plugin != nullptr) {
119  socketContextUpgradeFactory = plugin();
120 
121  if (socketContextUpgradeFactory != nullptr) {
122  if (SocketContextUpgradeFactorySelector::instance()->add(socketContextUpgradeFactory, handle)) {
123  VLOG(0) << "UpgradeSocketContext loaded successfully: " << filePath;
124  } else {
125  socketContextUpgradeFactory->destroy();
126  socketContextUpgradeFactory = nullptr;
127  dlclose(handle);
128  VLOG(0) << "UpgradeSocketContext already existing. Not using: " << filePath;
129  }
130  } else {
131  dlclose(handle);
132  VLOG(0) << "SocketContextUpgradeFactory not created: " << filePath;
133  }
134  } else {
135  dlclose(handle);
136  VLOG(0) << "Not a Plugin \"" << filePath;
137  }
138  } else {
139  VLOG(0) << "Error dlopen: " << dlerror();
140  }
141 
142  return socketContextUpgradeFactory;
143  }

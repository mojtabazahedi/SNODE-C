69  void listen(const SocketAddress& bindAddress, int backlog, const std::function<void(int)>& onError) {
70  Socket::open([this, &bindAddress, &backlog, &onError](int errnum) -> void {
71  if (errnum > 0) {
72  onError(errnum);
73  destruct();
74  } else {
75  reuseAddress([this, &bindAddress, &backlog, &onError](int errnum) -> void {
76  if (errnum != 0) {
77  onError(errnum);
78  destruct();
79  } else {
80  Socket::bind(bindAddress, [this, &backlog, &onError](int errnum) -> void {
81  if (errnum > 0) {
82  onError(errnum);
83  destruct();
84  } else {
85  int ret = net::system::listen(Socket::getFd(), backlog);
86 
87  if (ret == 0) {
88  AcceptEventReceiver::enable(Socket::getFd());
89  onError(0);

113  void acceptEvent() override {
114  typename SocketAddress::SockAddr remoteAddress{};
115  socklen_t remoteAddressLength = sizeof(remoteAddress);
116 
117  int fd = -1;
118 
119  fd = net::system::accept4(
120  Socket::getFd(), reinterpret_cast<struct sockaddr*>(&remoteAddress), &remoteAddressLength, SOCK_NONBLOCK);
121 
122  if (fd >= 0) {
123  typename SocketAddress::SockAddr localAddress{};
124  socklen_t addressLength = sizeof(localAddress);
125 
126  if (net::system::getsockname(fd, reinterpret_cast<sockaddr*>(&localAddress), &addressLength) == 0) {
127  SocketConnection* socketConnection = new SocketConnection(
128  fd, socketContextFactory, SocketAddress(localAddress), SocketAddress(remoteAddress), onConnect, onDisconnect);
129 
130  onConnected(socketConnection);
131  } else {
132  PLOG(ERROR) << "getsockname";
133  net::system::shutdown(fd, SHUT_RDWR);
134  net::system::close(fd);
135  }
136  } else if (errno != EINTR) {
137  PLOG(ERROR) << "accept";
138  }
139  }

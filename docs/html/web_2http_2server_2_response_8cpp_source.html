43  void Response::enqueue(const char* junk, std::size_t junkLen) {
44  if (!headersSent && !sendHeaderInProgress) {
45  sendHeaderInProgress = true;
46  sendHeader();
47  sendHeaderInProgress = false;
48  headersSent = true;
49  }
50 
51  serverContext->sendToPeer(junk, junkLen);
52 
53  if (headersSent) {
54  contentSent += junkLen;
55  if (contentSent == contentLength) {
56  serverContext->sendToPeerCompleted();
57  } else if (contentSent > contentLength) {
58  serverContext->terminateConnection();
59  }
60  }
61  }

113  Response& Response::set(const std::string& field, const std::string& value, bool overwrite) {
114  if (overwrite || headers.find(field) == headers.end()) {

117  if (field == "Content-Length") {

165  void Response::sendHeader() {
166  enqueue("HTTP/1.1 " + std::to_string(responseStatus) + " " + StatusCode::reason(responseStatus) + "\r\n");
167  enqueue("Date: " + httputils::to_http_date() + "\r\n");
168 
169  set({{"Cache-Control", "public, max-age=0"}, {"Accept-Ranges", "bytes"}, {"X-Powered-By", "snode.c"}});
170 
171  for (auto& [field, value] : headers) {
172  enqueue(field + ": " + value + "\r\n");
173  }
174 
175  for (auto& [cookie, cookieValue] : cookies) {
176  std::string cookieString =
177  std::accumulate(cookieValue.getOptions().begin(),
178  cookieValue.getOptions().end(),
179  cookie + "=" + cookieValue.getValue(),
180  [](const std::string& str, const std::pair<const std::string&, const std::string&> option) -> std::string {
181  return str + "; " + option.first + (!option.second.empty() ? "=" + option.second : "");
182  });
183  enqueue("Set-Cookie: " + cookieString + "\r\n");
184  }
185 
186  enqueue("\r\n");
187 
188  if (headers.find("Content-Length") != headers.end()) {
189  contentLength = std::stoul(headers.find("Content-Length")->second);
190  } else {
191  contentLength = 0;
192  }
193  }

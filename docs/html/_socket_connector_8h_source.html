72  void connect(const SocketAddress& remoteAddress, const SocketAddress& bindAddress, const std::function<void(int)>& onError) {
73  this->onError = onError;
74 
75  Socket::open(
76  [this, &bindAddress, &remoteAddress, &onError](int errnum) -> void {
77  if (errnum > 0) {
78  onError(errnum);
79  destruct();
80  } else {
81  Socket::bind(bindAddress, [this, &remoteAddress, &onError](int errnum) -> void {
82  if (errnum > 0) {
83  onError(errnum);
84  destruct();
85  } else {
86  int ret = net::system::connect(getFd(), &remoteAddress.getSockAddr(), remoteAddress.getSockAddrLen());
87 
88  if (ret == 0 || errno == EINPROGRESS) {
89  ConnectEventReceiver::enable(getFd());
90  } else {
91  onError(errno);
92  destruct();
93  }
94  }
95  });
96  }
97  },
98  SOCK_NONBLOCK);

102  void connectEvent() override {
103  int cErrno = -1;
104  socklen_t cErrnoLen = sizeof(cErrno);
105 
106  int err = net::system::getsockopt(Socket::getFd(), SOL_SOCKET, SO_ERROR, &cErrno, &cErrnoLen);
107 
108  if (err == 0) {
109  errno = cErrno;
110  if (errno != EINPROGRESS) {
111  if (errno == 0) {
112  typename SocketAddress::SockAddr localAddress{};
113  socklen_t localAddressLength = sizeof(localAddress);
114 
115  typename SocketAddress::SockAddr remoteAddress{};
116  socklen_t remoteAddressLength = sizeof(remoteAddress);
117 
118  if (net::system::getsockname(getFd(), reinterpret_cast<sockaddr*>(&localAddress), &localAddressLength) == 0 &&
119  net::system::getpeername(getFd(), reinterpret_cast<sockaddr*>(&remoteAddress), &remoteAddressLength) == 0) {
120  socketConnection = new SocketConnection(getFd(),
121  socketContextFactory,

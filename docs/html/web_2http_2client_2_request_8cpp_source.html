117  void Request::enqueue(const char* junk, std::size_t junkLen) {
118  if (!headersSent && !sendHeaderInProgress) {
119  sendHeaderInProgress = true;
120  sendHeader();
121  sendHeaderInProgress = false;
122  headersSent = true;
123  }
124 
125  clientContext->sendToPeer(junk, junkLen);
126 
127  if (headersSent) {
128  contentSent += junkLen;
129  if (contentSent == contentLength) {
130  clientContext->sendToPeerCompleted();
131  } else if (contentSent > contentLength) {
132  clientContext->terminateConnection();
133  }
134  }
135  }

141  void Request::sendHeader() {
142  std::string httpVersion = "HTTP/" + std::to_string(httpMajor) + "." + std::to_string(httpMinor);
143 
144  std::string queryString = "";
145  if (queries.size() > 0) {
146  queryString += "?";
147  for (auto& [key, value] : queries) {
148  queryString += httputils::url_encode(key) + "=" + httputils::url_encode(value) + "&";
149  }
150  queryString.pop_back();
151  }
152 
153  enqueue(method + " " + url + queryString + " " + httpVersion + "\r\n");
154 
155  enqueue("Host: " + host + "\r\n");
156  enqueue("Date: " + httputils::to_http_date() + "\r\n");
157 
158  headers.insert({{"Cache-Control", "public, max-age=0"}, {"Accept-Ranges", "bytes"}, {"X-Powered-By", "snode.c"}});
159 
160  for (const auto& [field, value] : headers) {
161  enqueue(field + ":" + value + "\r\n");
162  }
163 
164  if (contentLength != 0) {
165  enqueue("Content-Length: " + std::to_string(contentLength) + "\r\n");
166  }
167 
168  if (connectionState == ConnectionState::Keep) {
169  enqueue("Connection: keep-alive\r\n");
170  } else {
171  enqueue("Connection: close\r\n");
172  }
173 
174  for (const auto& [name, value] : cookies) {
175  enqueue("Cookie:" + name + "=" + value + "\r\n");
176  }
177 
178  enqueue("\r\n");
179 
180  if (headers.find("Content-Length") != headers.end()) {
181  contentLength = std::stoul(headers.find("Content-Length")->second);
182  } else {
183  contentLength = 0;
184  }
185  }

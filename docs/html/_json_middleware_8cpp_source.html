41  JsonMiddleware::JsonMiddleware() {
42  use([] MIDDLEWARE(req, res, next) {
43  try {
44  // parse body string with json library
45  // store it as type json from nlohmann library
46  nlohmann::json json = nlohmann::json::parse(req.body, req.body + req.contentLength);
47 
48  // set all the json data as attributes in the request object
49  req.setAttribute<nlohmann::json>(json);
50  } catch (nlohmann::detail::parse_error& error) {
51  // silently fail if body is not json
52  }
53 
54  next();
55  });
56  }

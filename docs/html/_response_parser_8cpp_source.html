64  enum Parser::ParserState ResponseParser::parseStartLine(const std::string& line) {
65  enum Parser::ParserState parserState = Parser::ParserState::HEADER;
66 
67  if (!line.empty()) {
68  std::string remaining;
69 
70  std::tie(httpVersion, remaining) = httputils::str_split(line, ' ');
71 
72  std::smatch httpVersionMatch;
73  if (!std::regex_match(httpVersion, httpVersionMatch, httpVersionRegex)) {
74  parserState = parsingError(400, "Wrong protocol version");
75  } else {
76  std::tie(statusCode, reason) = httputils::str_split(remaining, ' ');
77  if (StatusCode::contains(std::stoi(statusCode))) {
78  if (!reason.empty()) {
79  onResponse(httpVersion, statusCode, reason);
80  } else {
81  parserState = parsingError(400, "No reason phrase");
82  }
83  } else {
84  parserState = parsingError(400, "Unknown status code");
85  }
86  }
87  } else {
88  parserState = parsingError(400, "Response line empty");
89  }
90  return parserState;
91  }

93  enum Parser::ParserState ResponseParser::parseHeader() {
94  for (const auto& [field, value] : Parser::headers) {
95  if (field != "set-cookie") {
96  if (field == "content-length") {
97  Parser::contentLength = static_cast<std::size_t>(std::stoi(value));
98  }
99  } else {
100  std::string cookiesLine = value;
101 
102  while (!cookiesLine.empty()) {
103  std::string cookieLine;
104  std::tie(cookieLine, cookiesLine) = httputils::str_split(cookiesLine, ',');
105 
106  std::string cookieOptions;
107  std::string cookie;
108  std::tie(cookie, cookieOptions) = httputils::str_split(cookieLine, ';');
109 
110  std::string name;
111  std::string value;
112  std::tie(name, value) = httputils::str_split(cookie, '=');
113  httputils::str_trimm(name);
114  httputils::str_trimm(value);
115 
116  std::map<std::string, CookieOptions>::iterator cookieElement;
117  bool inserted;
118  std::tie(cookieElement, inserted) = cookies.insert({name, CookieOptions(value)});
119 
120  while (!cookieOptions.empty()) {
121  std::string option;
122  std::tie(option, cookieOptions) = httputils::str_split(cookieOptions, ';');
123 
124  std::string optionName;
125  std::string optionValue;
126  std::tie(optionName, optionValue) = httputils::str_split(option, '=');
127  httputils::str_trimm(optionName);
128  httputils::str_trimm(optionValue);
129 
130  cookieElement->second.setOption(optionName, optionValue);
131  }
132  }
133  }
134  }
135 
136  Parser::headers.erase("set-cookie");
137 
138  onHeader(Parser::headers, cookies);
139 
140  enum Parser::ParserState parserState = Parser::ParserState::BODY;
141  if (contentLength == 0) {
142  parsingFinished();
143  parserState = ParserState::BEGIN;
144  }
145 
146  return parserState;
147  }

51  switch (sslErr) {
52  case SSL_ERROR_WANT_WRITE:
53  LOG(INFO) << "SSL/TLS start renegotiation on read";
54  doSSLHandshake(
55  [](void) -> void {
56  LOG(INFO) << "SSL/TLS renegotiation on read success";
57  },
58  [](void) -> void {
59  LOG(WARNING) << "SSL/TLS renegotiation on read timed out";
60  },
61  [this](int sslErr) -> void {
62  ssl_log("SSL/TLS renegotiation", sslErr);
63  setSSLError(sslErr);
64  });
65  errno = EAGAIN;
66  break;
67  case SSL_ERROR_WANT_READ:
68  ret = 0;
69  errno = EAGAIN;
70  break;
71  case SSL_ERROR_ZERO_RETURN:
72  ret = 0;
73  errno = 0;
74  break;
75  case SSL_ERROR_SYSCALL:
76  if (errno != 0) {
77  ssl_log("SSL/TLS read failed", sslErr);
78  }
79  break;
80  default:

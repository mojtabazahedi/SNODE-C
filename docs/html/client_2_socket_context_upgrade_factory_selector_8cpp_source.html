81  void SocketContextUpgradeFactorySelector::registerSocketContextUpgradeFactory(
82  web::http::client::SocketContextUpgradeFactory* socketContextUpgradeFactory, void* handle) {
83  SocketContextPlugin socketContextPlugin = {.socketContextUpgradeFactory = socketContextUpgradeFactory, .handle = handle};
84 
85  if (socketContextUpgradeFactory->role() == SocketContextUpgradeFactory::ROLE::SERVER) {
86  const auto [it, success] = serverSocketContextPlugins.insert({socketContextUpgradeFactory->name(), socketContextPlugin});
87  if (!success) {
88  VLOG(0) << "UpgradeSocketContext already existing: not using " << socketContextUpgradeFactory->name();
89  socketContextUpgradeFactory->destroy();
90  if (handle != nullptr) {
91  dlclose(handle);
92  }
93  }
94  } else if (socketContextUpgradeFactory->role() == SocketContextUpgradeFactory::ROLE::CLIENT) {
95  const auto [it, success] = clientSocketContextPlugins.insert({socketContextUpgradeFactory->name(), socketContextPlugin});
96  if (!success) {
97  VLOG(0) << "UpgradeSocketContext already existing: not using " << socketContextUpgradeFactory->name();
98  socketContextUpgradeFactory->destroy();
99  if (handle != nullptr) {
100  dlclose(handle);
101  }
102  }
103  }

34  StaticMiddleware::StaticMiddleware(const std::string& root)
35  : root(root) {
36  use([&stdHeaders = this->stdHeaders, &stdCookies = this->stdCookies, &forceClose = this->forceClose] MIDDLEWARE(req, res, next) {
37  if (req.method == "GET") {
38  if (forceClose) {
39  res.set("Connection", "Close");
40  } else {
41  res.set("Connection", "Keep-Alive");
42  }
43  res.set(stdHeaders);
44  for (auto& [value, options] : stdCookies) {
45  res.cookie(value, options.getValue(), options.getOptions());
46  }
47  next();
48  } else {
49  LOG(DEBUG) << "Wrong method " << req.method;
50  res.set("Connection", "Close");
51  res.sendStatus(400);
52  }
53  });
54 
55  use([] MIDDLEWARE(req, res, next) {
56  if (req.url == "/") {
57  LOG(INFO) << "REDIRECT " + req.url + " -> " + "/index.html";
58  res.redirect(308, "/index.html");
59  } else {
60  next();
61  }
62  });

64  use([&root = this->root] APPLICATION(req, res) {
65  LOG(INFO) << "GET " + req.url + " -> " + root + req.url;
66  res.sendFile(root + req.url, [&req](int ret) -> void {
67  if (ret != 0) {
68  PLOG(ERROR) << req.url;
69  }
70  });
71  });

52  web::http::server::Request& req,
53  web::http::server::Response& res) {
54  std::string subProtocolName = req.header("sec-websocket-protocol");
55 
56  SocketContext* context = nullptr;
57 
58  SubProtocolFactory* subProtocolFactory =
59  dynamic_cast<SubProtocolFactory*>(SubProtocolFactorySelector::instance()->select(subProtocolName));
60 
61  if (subProtocolFactory != nullptr) {
62  if (subProtocolFactory->role() == SubProtocolFactory::Role::SERVER) {
63  SubProtocol* subProtocol = dynamic_cast<SubProtocol*>(subProtocolFactory->create());
64 
65  if (subProtocol != nullptr) {
66  subProtocol->setClients(subProtocolFactory->getClients());
67 
68  context = new SocketContext(socketConnection, subProtocol);
69 
70  if (context != nullptr) {
71  res.set("Upgrade", "websocket");
72  res.set("Connection", "Upgrade");
73  res.set("Sec-WebSocket-Protocol", subProtocolName);
74 
75  web::websocket::serverWebSocketKey(req.header("sec-websocket-key"), [&res](char* key) -> void {
76  res.set("Sec-WebSocket-Accept", key);
77  });
78 
79  res.status(101).end(); // Switch Protocol
80  } else {
81  subProtocolFactory->destroy(subProtocol);
82  res.status(500).end(); // Internal Server Error
83  }
84  } else {
85  res.status(404).end(); // Not Found
86  }
87  } else {

89  }

39  std::string url_decode(const std::string& text) {
40  int h;
41 
42  std::string escaped;
43 
44  for (std::string::const_iterator i = text.begin(), n = text.end(); i != n; ++i) {
45  std::string::value_type c = (*i);
46 
47  if (c == '%') {
48  if (i[1] && i[2]) {
49  h = from_hex(i[1]) << 4 | from_hex(i[2]);
50  escaped += static_cast<char>(h);
51  i += 2;
52  }
53  } else if (c == '+') {
54  escaped += ' ';
55  } else {
56  escaped += c;
57  }
58  }
59 
60  return escaped;
61  }

63  std::string url_encode(const std::string& text) {
64  std::ostringstream escaped;
65  escaped.fill('0');
66  escaped << std::hex;
67 
68  for (std::string::const_iterator i = text.begin(), n = text.end(); i != n; ++i) {
69  std::string::value_type c = (*i);
70 
71  if (isalnum(c) || c == '-' || c == '_' || c == '.' || c == '~') {
72  escaped << c;
73  } else {
74  escaped << std::uppercase;
75  escaped << '%' << std::setw(2) << int((unsigned char) c);
76  escaped << std::nouppercase;
77  }
78  }
79 
80  return escaped.str();
81  }

61  SSL_set_connect_state(ssl);

63  socketConnection->doSSLHandshake(

65  LOG(INFO) << "SSL/TLS initial handshake success";

68  [this](void) -> void { // onTimeout

70  this->onError(ETIMEDOUT);

72  [this](int sslErr) -> void { // onError
73  ssl_log("SSL/TLS initial handshake failed", sslErr);
74  this->onError(-sslErr);
75  });

55

103  std::size_t Parser::readHeaderLine() {
104  std::size_t consumed = 0;
105  std::size_t ret = 0;
106 
107  do {
108  char ch = 0;
109  ret = socketContext->readFromPeer(&ch, 1);
110 
111  if (ret > 0) {
112  if (ch == '\r' || ch == '\n') {
113  consumed++;
114  if (ch == '\n') {
115  if (EOL) {
116  splitHeaderLine(line);
117  line.clear();
118  if (parserState != ParserState::ERROR) {
119  parserState = parseHeader();
120  }
121  EOL = false;
122  } else {
123  if (line.empty()) {
124  if (parserState != ParserState::ERROR) {
125  parserState = parseHeader();
126  }
127  } else {
128  EOL = true;
129  }
130  }
131  }
132  } else if (EOL) {
133  if (std::isblank(ch)) {
134  if ((hTTPCompliance & HTTPCompliance::RFC7230) == HTTPCompliance::RFC7230) {
135  parserState = parsingError(400, "Header Folding");
136  } else {
137  line += ch;
138  consumed++;
139  }
140  } else {
141  splitHeaderLine(line);
142  line.clear();
143  line += ch;
144  consumed++;
145  }
146  EOL = false;
147  } else {
148  line += ch;
149  consumed++;
150  }
151  }
152  } while (ret > 0 && parserState == ParserState::HEADER);
153 
154  return consumed;

157  void Parser::splitHeaderLine(const std::string& line) {
158  if (!line.empty()) {
159  std::string field;
160  std::string value;
161  std::tie(field, value) = httputils::str_split(line, ':');
162 
163  if (field.empty()) {
164  parserState = parsingError(400, "Header-field empty");
165  } else if (std::isblank(field.back()) || std::isblank(field.front())) {
166  parserState = parsingError(400, "White space before or after header-field");
167  } else if (value.empty()) {
168  parserState = parsingError(400, "Header-value of field \"" + field + "\" empty");
169  } else {
170  httputils::to_lower(field);
171  httputils::str_trimm(value);
172 
173  if (headers.find(field) == headers.end()) {
174  headers.insert({field, value});
175  } else {
176  headers[field] += "," + value;
177  }
178  }
179  } else {
180  parserState = parsingError(400, "Header-line empty");
181  }

184  std::size_t Parser::readContent() {
185  if (contentRead == 0) {
186  content = new char[contentLength];
187  }
188 
189  std::size_t contentJunkLen =
190  (contentLength - contentRead < MAX_CONTENT_JUNK_LEN) ? contentLength - contentRead : MAX_CONTENT_JUNK_LEN;
191 
192  contentJunkLen = socketContext->readFromPeer(contentJunk, contentJunkLen);
193 
194  if (contentJunkLen > 0) {
195  if (contentRead + contentJunkLen <= contentLength) {
196  memcpy(content + contentRead, contentJunk, contentJunkLen); // NOLINT(clang-analyzer-core.NonNullParamChecker)
197 
198  contentRead += contentJunkLen;
199  if (contentRead == contentLength) {
200  parserState = parseContent(content, contentLength);
201 
202  delete[] content;
203  content = nullptr;
204  contentRead = 0;
205  }
206  } else {
207  parserState = parsingError(400, "Content to long");
208 
209  if (content != nullptr) {
210  delete[] content;
211  content = nullptr;
212  }
213  }
214  }
215 
216  return contentJunkLen;

74  void SocketContext::onFrameReceived(const char* junk, uint64_t junkLen) {
75  switch (opCodeReceived) {
76  case OpCode::CLOSE:
77  [[fallthrough]];
78  case OpCode::PING:
79  [[fallthrough]];
80  case OpCode::PONG:
81  pongCloseData += std::string(junk, static_cast<std::size_t>(junkLen));
82  break;
83  default:
84  std::size_t junkOffset = 0;
85 
86  do {
87  std::size_t sendJunkLen =
88  (junkLen - junkOffset <= SIZE_MAX) ? static_cast<std::size_t>(junkLen - junkOffset) : SIZE_MAX;
89  subProtocol->onMessageData(junk + junkOffset, sendJunkLen);
90  junkOffset += sendJunkLen;
91  } while (junkLen - junkOffset > 0);
92  break;
93  }
94  }

96  void SocketContext::onMessageEnd() {
97  switch (opCodeReceived) {
98  case OpCode::CLOSE:
99  if (closeSent) { // active close
100  closeSent = false;
101  VLOG(0) << "Close confirmed from peer";
102  } else { // passive close
103  VLOG(0) << "Close request received - replying with close";
104  sendClose(pongCloseData.data(), pongCloseData.length());
105  pongCloseData.clear();
106  }
107  break;
108  case OpCode::PING:
109  sendPong(pongCloseData.data(), pongCloseData.length());
110  pongCloseData.clear();
111  break;
112  case OpCode::PONG:
113  subProtocol->onPongReceived();
114  break;
115  default:
116  subProtocol->onMessageEnd();
117  break;
118  }
119  }

146  void SocketContext::sendClose(uint16_t statusCode, const char* reason, std::size_t reasonLength) {
147  char* closePayload = const_cast<char*>(reason);
148  std::size_t closePayloadLength = reasonLength;
149 
150  if (statusCode != 0) {
151  closePayload = new char[reasonLength + 2];
152  *reinterpret_cast<uint16_t*>(closePayload) = htobe16(statusCode);
153  closePayloadLength += 2;
154  if (reasonLength > 0) {
155  memcpy(closePayload + 2, reason, reasonLength);
156  }
157  }
158 
159  sendClose(closePayload, closePayloadLength);
160 
161  if (statusCode != 0) {
162  delete[] closePayload;
163  }
164 
165  setTimeout(CLOSE_SOCKET_TIMEOUT);
166 
167  closeSent = true;
168  }

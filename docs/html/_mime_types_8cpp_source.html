221  MimeTypes::MimeTypes() {
222  MimeTypes::magic = magic_open(MAGIC_MIME);
223 
224  if (magic_load(magic, nullptr) != 0) {
225  LOG(WARNING) << "cannot load magic database - " + std::string(magic_error(magic));
226  magic_close(magic);
227  magic = nullptr;
228  }
229  }

235  std::string MimeTypes::contentType(const std::string& file) {
236  std::map<std::string, std::string>::iterator it = mimeType.find(std::filesystem::path(file).extension());
237 
238  std::string type;
239 
240  if (it != mimeType.end()) {
241  type = it->second;
242  } else if (magic != nullptr) {
243  type = magic_file(MimeTypes::magic, file.c_str());
244  }
245 
246  return type;
247  }
